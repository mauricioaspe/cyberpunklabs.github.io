all_samples = pickle.load(open('data_bases/all.samples2', 'rb'))

df1 = pd.DataFrame({'mu': np.mean(all_samples['agotados']['mu'], axis=(0,2))})
df1['condicion'] = 'EE (+)'

df2 = pd.DataFrame({'mu': np.mean(all_samples['no_agotados']['mu'], axis=(0,2))})
df2['condicion'] = 'EE (-)'

agotados = df1.append(df2)

df1 = pd.DataFrame({'mu': np.mean(all_samples['no_quemados']['mu'], axis=(0,2))})
df1['condicion'] = 'Burnout (+)'

df2 = pd.DataFrame({'mu': np.mean(all_samples['quemados']['mu'], axis=(0,2))})
df2['condicion'] = 'Burnout (-)'

quemados = df1.append(df2)

#########
p = 95

means = dict()
cis = dict()
for condition in all_samples.keys():  
    values = all_samples[condition]['mu']
    
    means[condition] = np.mean(values)
    cis[condition] = np.percentile(values, [100-p, p])
